import pandas as pd
from sklearn.model_selection import train_test_split
from sklearn.ensemble import RandomForestRegressor
from sklearn.metrics import mean_squared_error
import numpy as np

# Sample data loading (assuming CSV format)
# Replace 'your_dataset.csv' with your actual dataset file
df = pd.read_csv('your_dataset.csv')

# Example columns: Date, Total_Revenue, Operating_Costs, Profit_Margin, Cash_Inflows, Cash_Outflows
# Preprocessing and feature engineering
df['Date'] = pd.to_datetime(df['Date'])
df.set_index('Date', inplace=True)

# Assuming we want to forecast 'Cash_Outflows'
X = df.drop('Cash_Outflows', axis=1)  # Features
y = df['Cash_Outflows']  # Target variable

# Splitting the dataset into the Training set and Test set
X_train, X_test, y_train, y_test = train_test_split(X, y, test_size=0.2, random_state=0)

# Model building - Using Random Forest Regressor
model = RandomForestRegressor(n_estimators=100, random_state=0)
model.fit(X_train, y_train)

# Predicting the Test set results
y_pred = model.predict(X_test)

# Evaluating the Model
mse = mean_squared_error(y_test, y_pred)
rmse = np.sqrt(mse)

print("Model Evaluation:")
print("Mean Squared Error: ", mse)
print("Root Mean Squared Error: ", rmse)
